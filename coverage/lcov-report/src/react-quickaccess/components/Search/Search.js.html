import {withAppContext} from "../../contexts/AppContext";
 
class Search extends React.Component {
 
  constructor(props) {
    super(props);
    this.handleInputChange = this.handleInputChange.bind(this);
    this.searchInputRef = React.createRef();
  }
 
  /**
   * Get the translate function
