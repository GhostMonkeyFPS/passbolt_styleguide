  getUsersMatch(users, keyword) {
    const words = (keyword && keyword.split(/\s+/)) || [''];

    const escapeWord = word => word.replace(/[.*+?^${}()|[\]\\]/g, "\\$&");

    const matchWord = (word, value) => wordToRegex(word).test(value);
